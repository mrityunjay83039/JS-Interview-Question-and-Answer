Write a Javascript program to check two numbers and return true if one of the number is 100 or if sum of the two numbers is 100 

        
            function checkTwoNumbers(a, b){
                if((a == 100 || b == 100) || (a + b == 100) ){
                    return true;
                }
            }
            console.log(checkTwoNumbers(100, 2))   // true
            console.log(checkTwoNumbers(25, 75))  // true
